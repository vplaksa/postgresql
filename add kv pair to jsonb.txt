/*
Add key/value pair to jsonb*/


DO $$
DECLARE 
v_result jsonb;
BEGIN

SELECT json_object_agg(key, value)
INTO v_result
FROM
(
SELECT key, value from jsonb_each('{"a": "valA", "b": "valB"}'::jsonb)
UNION ALL
SELECT 'c' AS key, '"valC"'::jsonb AS value) AS combined;

RAISE NOTICE '%', v_result::text;
END;
$$
